# 追加偶数

```rust
fn generate_even_number_report() {
    let data = vec![1, 2, 3, 4, 5, 6, 7, 8];
    
    // 报告的初始状态是一个可变的空字符串
    let mut report = String::from("偶数报告:\n");

    // 遍历数据
    for num in data {
        if num % 2 == 0 {
            // 将偶数的信息追加到报告字符串中
            // to_string() 会把数字转换成字符串
            report.push_str(&num.to_string());
            report.push_str("\n"); // 加个换行符
        }
    }

    println!("{}", report);
}

// 调用函数
// generate_even_number_report();
```
