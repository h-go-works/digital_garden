# 追加偶数

```rust
fn generate_even_number_report() {
    let data = vec![1, 2, 3, 4, 5, 6, 7, 8];
    
    // 报告的初始状态是一个可变的空字符串
    let mut report = String::from("偶数报告:\n");

    // 遍历数据
    for num in data {
        if num % 2 == 0 {
            // 将偶数的信息追加到报告字符串中
            // to_string() 会把数字转换成字符串
            report.push_str(&num.to_string());
            report.push_str("\n"); // 加个换行符
        }
    }

    println!("{}", report);
}

// 调用函数
// generate_even_number_report();
```


# 不能被赋值因为是let

```rust
let mut s1 = String::from("A");
let mut s2 = String::from("B");

// my_ref 这个“绑定”是不可变的
let my_ref = &mut s1; 

// 错误！你不能让 my_ref 这个变量转而去指向 s2
// my_ref = &mut s2;
```


# 可以被赋值因为是let mut

```rust
let mut s1 = String::from("A");
let mut s2 = String::from("B");

// my_ref 这个“绑定”是可变的
let mut my_ref = &mut s1; 

// 完全可以！现在 my_ref 从指向 s1 转为指向 s2
my_ref = &mut s2; 

// 现在修改的是 s2
my_ref.push_str("!");
println!("{}", s2); // 打印 "B!"
```


# 通过 &mut 引用，去彻底替换“它指向的那个值

```rust
let mut s = String::from("Old value");

let my_ref = &mut s;

// 使用解引用 `*`，可以彻底替换掉 s 的值
*my_ref = String::from("New value");

println!("{}", s); // 打印 "New value"
```
